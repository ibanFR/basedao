3   import com.ibanfr.domain.model.Employee;
4   import com.ibanfr.domain.model.Team;
5   import jakarta.enterprise.context.ApplicationScoped;
6   import jakarta.enterprise.inject.Produces;
7   import org.hibernate.SessionFactory;
8   import org.hibernate.boot.Metadata;
9   import org.hibernate.boot.MetadataSources;
10  import org.hibernate.boot.registry.StandardServiceRegistryBuilder;
11  import org.hibernate.service.ServiceRegistry;
12  
13  import java.util.Map;
14  
15  /**
16   *
17   * Uses hibernate native bootstrapping mechanism
18   *
19   * <a href="https://docs.jboss.org/hibernate/orm/5.4/topical/html_single/bootstrap/NativeBootstrapping.html">Native bootstrapping</a>
20   */
21  public class SessionFactoryProducer {
22  
23      @Produces
24      @ApplicationScoped
25      SessionFactory produceH2SessionFactory(){
26          Map<String, Object> settings = Map.of("hibernate.connection.url", "jdbc:h2:mem:testdb",
27                                "hibernate.connection.driver_class", "org.h2.Driver",
28                                "hibernate.hbm2ddl.auto", "update",
29                                "hibernate.show_sql", "false",
30                                "hibernate.current_session_context_class","thread");
31  
32          return getSessionFactory(settings, Employee.class, Team.class);
33      }
34  
35      /**
36       * Produces a SessionFactory instance.
37       * <p>
38       * The SessionFactory instance is created using the specified settings and annotated classes.
39       *
40       * @param settings
41       * @param annotatedClasses
42       * @return
43       */
44      protected SessionFactory produceSessionFactory(Map<String, Object> settings,
45                                                     Class<?>... annotatedClasses) {
46          return getSessionFactory(settings, annotatedClasses);
47      }
48  
49      /**
50       * Produces a SessionFactory instance.
51       * <p>
52       * The SessionFactory instance is created using the specified settings and annotated classes.
53       *
54       * @param settings
55       * @param annotatedClasses
56       * @return
57       */
58      SessionFactory getSessionFactory(Map<String, Object> settings, Class<?>... annotatedClasses){
59  
60          //Build ServiceRegistry
61          ServiceRegistry serviceRegistry = buildServiceRegistry(settings);
62          //Create the Metadata object using the specified ServiceRegistry.
63          Metadata metadata = buildMetadata(serviceRegistry, annotatedClasses);
64          //Build the SessionFactory
65          return buildSessionFactory(metadata);
66      }
67  
68      /**
69       * Build the ServiceRegistry instance.
70       * <p>
71       * See
72       * <a href="https://docs.jboss.org/hibernate/orm/current/userguide/html_single/Hibernate_User_Guide.html#bootstrap-native-ServiceRegistry">Hibernate Employee Guide#Building the ServiceRegistry</a>
73       *
74       * @param settings
75       * @return
76       */
77      ServiceRegistry buildServiceRegistry(Map<String, Object> settings) {
78  
79          //Build ServiceRegistry
80          ServiceRegistry serviceRegistry;
81          serviceRegistry = new StandardServiceRegistryBuilder()/* .configure() */
82                                                                .applySettings(settings)
83                                                                .build();
84          return serviceRegistry;
85      }
86  
87      /**
88       * Create a Metadata object containing the parsed representations of the application domain model and its mapping
89       * to the database.
90       * <p>
91       * See
92       * <a href="https://docs.jboss.org/hibernate/orm/current/userguide/html_single/Hibernate_User_Guide.html#bootstrap-native-metadata">Hibernate Employee Guide#Building the Metadata</a>
93       *
94       * @param serviceRegistry
95       * @return
96       */
97      Metadata buildMetadata(ServiceRegistry serviceRegistry, Class<?>... annotatedClasses){
98          Metadata metadata;
99          metadata = new MetadataSources(serviceRegistry).addAnnotatedClasses(annotatedClasses)
100                                                        .getMetadataBuilder()
101                                                        .build();
102         return metadata;
103     }
104 
105     /**
106      * Build the SessionFactory instance.
107      * <p>
108      * See
109      * <a href="https://docs.jboss.org/hibernate/orm/current/userguide/html_single/Hibernate_User_Guide.html#bootstrap-native-SessionFactory">Hibernate Employee Guide#Building the SessionFactory</a>
110      *
111      * @param metadata
112      * @return
113      */
114     SessionFactory buildSessionFactory(Metadata metadata){
115         SessionFactory factory;
116         factory = metadata.getSessionFactoryBuilder()
117                           .build();
118         return factory;
119     }
120 }
